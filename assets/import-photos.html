Import Instagram photos for Koshi Chhyang

Console showed undefined? That is normal. copy() does not return a value — check your clipboard (paste into Notepad). The script below also prints how many URLs it found.

  1. Log into instagram.com/koshichhyang.
  2. Scroll down the profile grid so several rows of posts load.
  3. F12 → Console → paste Snippet A → Enter.

Snippet A (copy + print)

(() => {
  const pick = (img) => {
    const src = img.currentSrc || img.src || "";
    if (src && /cdninstagram|fbcdn\.net/.test(src)) return src;
    const best = (img.srcset || "").split(",").map((s) => s.trim().split(" ")[0]).filter(Boolean).pop();
    return best || "";
  };
  const urls = [...document.querySelectorAll("img")]
    .map(pick)
    .filter((u) => u && /cdninstagram|fbcdn\.net/.test(u))
    .filter((u) => !/150x150|s150x150|profile_pic|emoji/.test(u))
    .filter((u, i, a) => a.indexOf(u) === i)
    .slice(0, 8);
  console.log("Found " + urls.length + " image URL(s):");
  urls.forEach((u, i) => console.log((i + 1) + ". " + u));
  if (!urls.length) {
    console.warn("No URLs — scroll the profile grid or use Snippet B on one open post.");
    return urls;
  }
  copy(urls.join("\n"));
  console.log("Copied to clipboard. Paste into assets/image-urls.txt");
  return urls;
})()

If Snippet A says “Found 0” — Snippet B (one post)

  1. Click any post on the profile so it opens full-screen.
  2. Run this in the console:
(() => {
  const urls = [...document.querySelectorAll('img, [style*="background-image"]')]
    .flatMap((el) => {
      if (el.tagName === "IMG") {
        const u = el.currentSrc || el.src;
        return u ? [u] : [];
      }
      const m = (el.style.backgroundImage || "").match(/url\("(.+?)"\)/);
      return m ? [m[1]] : [];
    })
    .filter((u) => /cdninstagram|fbcdn\.net/.test(u))
    .filter((u, i, a) => a.indexOf(u) === i);
  console.log(urls);
  copy(urls.slice(0, 3).join("\n"));
  return urls;
})()

Repeat for 2–3 posts if you need more than three URLs; paste all lines into image-urls.txt.
