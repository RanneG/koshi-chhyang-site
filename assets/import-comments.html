Import Instagram comments for Community notes

Instagram blocks automated scraping. Use this while logged in on @koshichhyang to copy real @handles, comment text, and the post image each comment belongs to.

  1. Open one post on @koshichhyang (comments visible).
  2. F12 → Console → paste Snippet C → Enter.
  3. Set the post key in the printed JSON (see table above).
  4. Paste the object into assets/community-comments-export.json (array of exports).
  5. Run: python scripts/merge_community_comments.py
  6. Refresh the landing page (Ctrl+F5).

Snippet C — one open post

(() => {
  const POST_HINTS = [
    ["hero", "509109336"],
    ["pour", "514470029"],
    ["product-1", "561102938"],
    ["product-2", "655996050"],
    ["product-3", "564908787"],
    ["brew", "688430214"],
    ["event", "505407617"],
  ];
  let POST_KEY = "pour"; // fallback if auto-detect misses
  const imgs = [...document.querySelectorAll("article img, main img, div[role=presentation] img")]
    .map((i) => i.currentSrc || i.src || "")
    .filter((u) => /cdninstagram|fbcdn\\.net/.test(u) && !/150x150|s150x150|profile_pic|emoji/.test(u))
    .sort((a, b) => b.length - a.length);
  const postImage = imgs[0] || "";
  for (const [key, id] of POST_HINTS) {
    if (postImage.includes(id)) { POST_KEY = key; break; }
  }
  const comments = [];
  const seen = new Set();
  document.querySelectorAll('a[href^="/"]').forEach((a) => {
    const m = (a.getAttribute("href") || "").match(/^\\/([A-Za-z0-9._]+)\\/?$/);
    if (!m) return;
    const handle = m[1];
    if (handle === "koshichhyang" || handle === "explore" || handle === "p" || handle === "reels") return;
    const row = a.closest("li") || a.closest('div[class*="Comment"]') || a.closest("div");
    if (!row || seen.has(handle + row.innerText.slice(0, 40))) return;
    const textEl = row.querySelector("span[dir=auto]") || row.querySelector("span");
    const text = textEl ? textEl.innerText.trim() : "";
    if (!text || text.length <= 2) return;
    seen.add(handle + text.slice(0, 40));
    const nameEl = row.querySelector("span:not([dir=auto])") || a;
    const name = (nameEl && nameEl !== textEl ? nameEl.innerText.trim().split("\\n")[0] : "") || handle;
    comments.push({ handle, name, text });
  });
  const out = { post: POST_KEY, postImage, comments: comments.slice(0, 12) };
  console.log(JSON.stringify(out, null, 2));
  copy(JSON.stringify(out, null, 2));
  console.log("Copied " + out.comments.length + " comment(s) for post \"" + POST_KEY + "\"");
  return out;
})()

Repeat for each post you want in the carousel. Append each exported object to the array in community-comments-export.json.
